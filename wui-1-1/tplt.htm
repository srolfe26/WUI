Template Engine

// Some random data
var data =	[
				{id:1, fname:'Steve'},
				{id:2, fname:'Dave'},
				{id:3, fname:'Dan'},
				{id:4, fname:'Sam'}
			];

// Create the template engine
a = new Wui.tplt({
	tplt:	'<li>Name: {fname}, conditional expression: {((id == 3)?"This has an ID of 3!!":fname)}</li>'
});

// Display something on the body
var target = $('<ul>').prependTo('body');

for(var i in data){
	a.data = data[i];
	target.append(a.make());
}
